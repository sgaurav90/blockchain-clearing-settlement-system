<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Ethereum Token Transfer</title>

</head>

<body>

    <h1>Ethereum Token Transfer</h1>

    <form id="transfer-form">

        <label for="to">Recipient Address:</label>

        <input type="text" id="to" name="to" required>

        <br>

        <label for="amount">Amount (in tokens):</label>

        <input type="number" id="amount" name="amount" required>

        <br>

        <button type="submit">Transfer</button>

    </form>

    <p id="result"></p>

    <script>

        document.getElementById('transfer-form').addEventListener('submit', async (event) => {

            event.preventDefault();

            const to = document.getElementById('to').value;

            const amount = document.getElementById('amount').value;


            try {

                const response = await fetch('/transfer', {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({ to, amount })

                });

                const result = await response.json();

                document.getElementById('result').innerText = `Transaction Hash: ${result.transactionHash}`;

            } catch (error) {

                document.getElementById('result').innerText = `Error: ${error.message}`;

            }

        });

    </script>

</body>

</html>
